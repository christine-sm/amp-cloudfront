let vs. var

  • Both let & var can be redefined
  • Variables declared with var are function scoped
  • Variables declared with let are block scoped {}

function scopeExample() {
  // function scoped
}

if () {
  // block scoped
}

var example


function varExample() {
  var centimeters = 150;
  if (centimeters > 100) {
    var meters = centimeters / 100;
  }
  return meters;
}
  • meters is defined
  • Temporary variable has leaked out of scope

let example


function letExample() {
  let centimeters = 150;
  if (centimeters > 100) {
    let meters = centimeters / 100;
  }
  return meters;
}
  • meters is undefined
  • Temporary variable has not leaked out of scope

const

  • The value of const is never reassigned
  • Does not necessarily mean than const is immutable
  • Properties of const can change
  • Use Object.freeze() if you want to make an immutable object

const exampleArray = [1, 2, 3];

const exampleObject = {
	month: 'December',
	days: 31
};

let vs. const

`${Template Strings}`

  • Use `` instead of ''
  • Use ${} for data interpolation

const fullName = `Slim Shady`;

// ES5 Template String
var oldString = 'My name is ' + fullName;

// ES6 Template String
const newString = `My name is ${fullName}`;

Arrow Functions

() =>

  • Does not rebind value of this; this bound to parent scope
  • Remember: traditional anonymous functions bind this to element

// ES5 Anonymous Function
var exampleButtonOne = document.querySelector('#exampleButtonOne');

exampleButtonOne.addEventListener('click', function() {
  console.log(this);
});

// ES6 Arrow Function
const exampleButtonTwo = document.querySelector('#exampleButtonTwo');

exampleButtonTwo.addEventListener('click', () => {
  console.log(this);
});

Arrow Functions (cont.)

  • Concise syntax
  • Implicit return for unbracketed functions

// Make it an ES6 arrow function!

function introduceMe (name, job) {
  return 'My name is ' + name + ' and I am a(n) ' + job;
}

Default Function Parameters

  • Function parameters can be assigned a default value


const introduceYou = (name, job, vowel = false) => {
  let article;
  if (vowel) {
    article = 'an';
  } else if (!vowel) {
    article = 'a';
  }
  return `My name is ${name} and I am ${article} ${job}`;
}
